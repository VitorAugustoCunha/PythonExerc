Create database escola;
Use escola;

CREATE TABLE sala1 (
	ID INT NOT NULL AUTO_INCREMENT,
	nome VARCHAR(255) NOT NULL,
	nota1 DOUBLE NOT NULL,
	nota2 DOUBLE NOT NULL,
	nota3 DOUBLE NOT NULL,
	media DOUBLE,
	PRIMARY KEY (ID)
);

CREATE TABLE sala2 (
	ID INT NOT NULL AUTO_INCREMENT,
	nome VARCHAR(255) NOT NULL,
	nota1 DOUBLE NOT NULL,
	nota2 DOUBLE NOT NULL,
	nota3 DOUBLE NOT NULL,
	media DOUBLE,
	FOREIGN KEY (ID) REFERENCES sala1(ID)
);
